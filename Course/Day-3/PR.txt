        let pricejeruk = 5000
        let pricemangga = 5000
        let priceleci = 5000
        let priceapel = 5000

        let stockmangga = 20
        let stockleci = 20
        let stockapel = 10

    Mangga Rp. ${pricemangga}/pcs (${stockmangga} item dalam stock)
    Leci Rp. ${priceleci}/pcs (${stockleci} item dalam stock)
    Apel Rp. ${priceapel}/pcs (${stockapel} item dalam stock)`)


            alert('Silakan Masukkan Berapa Buah yang ingin anda beli')
            let condition;
            do {
                var belijeruk = prompt('Berapa Buah Jeruk yang ingin anda beli ?') * 1
                if (belijeruk > stockjeruk) {
                    alert('Stock yang anda minta melebihi batas')
                    condition = true
                }
                else {
                    condition = false
                }
            }
            while (condition) {
            }


            do {
                var belimangga = prompt('Berapa Buah Mangga yang ingin anda beli ?') * 1
                if (belimangga > stockmangga) {
                    alert('Stock yang anda minta melebihi batas')
                    condition = true
                }
                else {
                    condition = false
                }
            }
            while (condition) {
            }

            do {
                var belileci = prompt('Berapa Buah Leci yang ingin anda beli ?') * 1
                if (belileci > stockleci) {
                    alert('Stock yang anda minta melebihi batas')
                    condition = true
                }
                else {
                    condition = false
                }
            }
            while (condition) {
            }

            alert('Jumlah yang harus anda bayar ' + totalharga)

                let uanganda = parseInt(prompt('Masukkan Uang Anda = '))
